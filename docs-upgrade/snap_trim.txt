#!/usr/bin/env bash
# snap-trim.sh V1.7
# 支持 --keep 0，并在危险操作时提示确认
#改为使用json格式输出快照ID，避免切分失效

set -o pipefail

DRY_RUN=0
RETRY_INTERVAL=5
POST_DELETE_SLEEP=2
KEEP_COUNT=1

usage() {
  cat <<EOF
Usage: $0 [--dry-run|-n] [--keep|-k N] disk_id_file
  disk_id_file: 文本文件，每行一个 RBD 镜像名（支持 # 注释和空行）
  --dry-run, -n : 仅显示操作，不实际删除
  --keep N, -k N: 保留最新的 N 个有效快照（N ≥ 0；N=0 表示删除所有符合条件的快照）
EOF
  exit 1
}

# === 参数解析：允许 N >= 0 ===
while [[ $# -gt 0 ]]; do
  case "$1" in
    -n|--dry-run)
      DRY_RUN=1
      shift
      ;;
    -k|--keep)
      if [[ -z "$2" || ! "$2" =～ ^[0-9]+$ ]] || (( $2 < 0 )); then
        echo "ERROR: --keep 需要一个 ≥0 的整数" >&2
        exit 1
      fi
      KEEP_COUNT="$2"
      shift 2
      ;;
    *)
      break
      ;;
  esac
done

if [[ $# -ne 1 ]]; then
  usage
fi
DISK_FILE="$1"

# === 依赖检查 ===
for cmd in ceph rbd jq awk sed sort date; do
  if ! command -v "$cmd" >/dev/null 2>&1; then
    echo "ERROR: 缺少必要命令: '$cmd'" >&2
    exit 2
  fi
done

# === 辅助函数 ===
get_pools() {
  if ceph osd lspools >/dev/null 2>&1; then
    ceph osd lspools 2>/dev/null | awk '{print $2}'
  else
    ceph osd pool ls 2>/dev/null
  fi
}

check_pgs_ok() {
  local out states
  out=$(ceph -s 2>/dev/null) || return 1
  states=$(echo "$out" | awk '/pgs:/,/^$/' | grep -o 'active[^ ,;]*')
  [[ -z "$states" ]] && return 1
  for st in $states; do
    [[ "$st" == "active+clean" || "$st" == "active+clean+scrubbing" ]] || return 1
  done
  return 0
}

wait_for_pgs_ok() {
  while ! check_pgs_ok; do
    echo "INFO: PG 状态不满足，等待 ${RETRY_INTERVAL}s..."
    sleep "$RETRY_INTERVAL"
  done
}

# === 主逻辑开始 ===
[[ ! -f "$DISK_FILE" ]] && { echo "ERROR: 文件不存在: $DISK_FILE" >&2; exit 3; }

POOLS=()
while IFS= read -r p || [[ -n "$p" ]]; do
  p=$(echo "$p" | awk '{print $1}')
  [[ -n "$p" ]] && POOLS+=("$p")
done < <(get_pools)

(( ${#POOLS[@]} == 0 )) && { echo "ERROR: 未发现任何存储池"; exit 4; }
echo "INFO: 存储池: ${POOLS[*]}"
echo "INFO: 保留最新 $KEEP_COUNT 个有效快照（名称≤15字符、未保护）"

# === ⚠️ 安全确认：当 KEEP_COUNT=0 且非 dry-run 时 ===
if (( KEEP_COUNT == 0 && DRY_RUN == 0 )); then
  echo ""
  echo "⚠️  警告：你选择了保留 0 个快照！"
  echo "   所有符合条件的快照（名称≤15字符、未保护）将被永久删除。"
  echo "   此操作不可逆！"
  read -p "确定继续吗？请输入 'yes' 确认: " confirm
  if [[ "$confirm" != "yes" ]]; then
    echo "❌ 操作已取消。"
    exit 0
  fi
  echo ""
fi

# === 处理每个磁盘 ID ===
while IFS= read -r line || [[ -n "$line" ]]; do
  disk_id=$(echo "$line" | sed 's/^[[:space:]]*//;s/[[:space:]]*$//')
  [[ -z "$disk_id" || "${disk_id:0:1}" == "#" ]] && continue
  echo "-----------------------------------------"
  echo "处理磁盘 id: $disk_id"

  found=0
  for pool in "${POOLS[@]}"; do
    if rbd info -p "$pool" "$disk_id" >/dev/null 2>&1; then
      found=1
      echo "INFO: 在 pool '$pool' 中找到镜像 '$disk_id'"

      # 获取快照列表（JSON 格式）
      snap_json=$(rbd snap ls --format=json -p "$pool" "$disk_id" 2>/dev/null)

      # 验证是否为有效 JSON 数组
      if ! echo "$snap_json" | jq 'type == "array"' >/dev/null 2>&1; then
        echo "WARN: 无法获取有效快照列表（非 JSON 或空），跳过"
        break
      fi

      tmpfile=$(mktemp) || { echo "ERROR: 无法创建临时文件" >&2; break; }

      # 解析每个快照
      echo "$snap_json" | jq -c '.[]' | while read -r snap; do
        snap_name=$(echo "$snap" | jq -r '.name // ""')
        protected=$(echo "$snap" | jq -r '.protected // false')
        timestamp_str=$(echo "$snap" | jq -r '.timestamp // ""')

        [[ -z "$snap_name" || -z "$timestamp_str" ]] && continue

        if [[ "$protected" == "true" ]]; then
          echo "INFO: 快照受保护，跳过: $snap_name"
          continue
        fi

        if (( ${#snap_name} > 15 )); then
          echo "INFO: 快照名过长（>15字符），忽略: $snap_name"
          continue
        fi

        if unixtime=$(date -d "$timestamp_str" +%s 2>/dev/null); then
          echo "$snap_name|$timestamp_str|$unixtime" >> "$tmpfile"
        else
          echo "WARN: 无法解析时间戳: $snap_name ($timestamp_str)" >&2
        fi
      done

      if [[ ! -s "$tmpfile" ]]; then
        echo "INFO: 无有效可清理快照，跳过"
        rm -f "$tmpfile"
        break
      fi

      mapfile -t sorted_lines < <(sort -t'|' -k3,3n "$tmpfile")
      rm -f "$tmpfile"

      total=${#sorted_lines[@]}
      if (( total <= KEEP_COUNT )); then
        echo "INFO: 有效快照数（$total）≤ 保留数（$KEEP_COUNT），全部保留"
        break
      fi

      snaps_to_delete=()
      delete_count=$((total - KEEP_COUNT))
      for ((i=0; i<delete_count; i++)); do
        snap_name="${sorted_lines[i]%%|*}"
        snaps_to_delete+=("$snap_name")
      done

      echo "INFO: 将删除 $delete_count 个旧快照（保留最新 $KEEP_COUNT 份）:"
      printf "  %s\n" "${snaps_to_delete[@]}"

      for snap in "${snaps_to_delete[@]}"; do
        img_spec="$pool/$disk_id@$snap"
        if (( DRY_RUN )); then
          echo "[DRY-RUN] rbd snap rm \"$img_spec\""
        else
          echo "执行: rbd snap rm \"$img_spec\""
          if rbd snap rm "$img_spec"; then
            echo "删除成功: $snap"
          else
            echo "ERROR: 删除失败: $snap" >&2
            break
          fi
        fi

        sleep "$POST_DELETE_SLEEP"

        if (( !DRY_RUN )); then
          echo "INFO: 检查 PG 状态..."
          wait_for_pgs_ok
        fi
      done
      break
    fi
  done

  (( !found )) && echo "WARN: 未找到镜像 $disk_id"
done < "$DISK_FILE"

echo "-----------------------------------------"
echo "完成。"
exit 0
